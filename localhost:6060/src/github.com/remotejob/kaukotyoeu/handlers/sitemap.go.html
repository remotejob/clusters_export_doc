    45	func CheckServeSitemap(w http.ResponseWriter, r *http.Request) {
    46	
    47		sitefull := r.Host
    48		site := strings.Split(sitefull, ":")[0]
    49	
    50		if site == "localhost" {
    51	
    52			site = "www.kaukotyo.eu"
    53	
    54		}
    55	
    56		mongoDBDialInfo := &mgo.DialInfo{
    57			Addrs:     addrs,
    58			Timeout:   60 * time.Second,
    59			Database:  database,
    60			Username:  username,
    61			Password:  password,
    62			Mechanism: mechanism,
    63		}
    64	
    65		dbsession, err := mgo.DialWithInfo(mongoDBDialInfo)
    66	
    67		if err != nil {
    68			panic(err)
    69		}
    70		defer dbsession.Close()
    71	
    72		allsitemaplinks := dbhandler.GetAllSitemaplinks(*dbsession, site)
    73	
    74		// fmt.Println("CheckServeSitemap")
    75	
    76		docList := new(domains.Pages)
    77		docList.XmlNS = "http://www.sitemaps.org/schemas/sitemap/0.9"
    78	
    79		for _, sitemaplink := range allsitemaplinks {
    80	
    81			if sitemaplink.Site == site {
    82	
    83				doc := new(domains.Page)
    84				doc.Loc = "http://" + site + "/" + themes + "/" + locale + "/" + mainroute + "/" + sitemaplink.Stitle + ".html"
    85				doc.Lastmod = sitemaplink.Updated.Format(time.RFC3339)
    86				doc.Changefreq = "monthly"
    87				docList.Pages = append(docList.Pages, doc)
    88				// fmt.Println(site, sitemaplink.Stitle)
    89			}
    90	
    91		}
    92	
    93		resultXML, err = xml.MarshalIndent(docList, "", "  ")
    94		if err != nil {
    95			log.Println(err.Error())
    96		}
    97	
    98		// fmt.Println(bytes.NewBuffer(resultXML).String())
    99	
   100		// fmt.Println(bytes.NewBuffer(sitemap).String())
   101	
   102		w.Header().Add("Content-type", "application/xml")
   103		w.Write(resultXML)
   104	
   105	}
