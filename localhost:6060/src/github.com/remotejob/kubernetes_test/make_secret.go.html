    14	// A small script that converts the given open ssl public/private keys to
    15	// a secret that it writes to stdout as json. Most common use case is to
    16	// create a secret from self signed certificates used to authenticate with
    17	// a devserver. Usage: go run make_secret.go -crt ca.crt -key priv.key > secret.json
    18	package main
    19	
    20	import (
    21		"flag"
    22		"fmt"
    23		"io/ioutil"
    24		"log"
    25	
    26		"k8s.io/kubernetes/pkg/api"
    27		"k8s.io/kubernetes/pkg/apimachinery/registered"
    28		"k8s.io/kubernetes/pkg/runtime"
    29	
    30		// This installs the legacy v1 API
    31		_ "k8s.io/kubernetes/pkg/api/install"
    32	)
    33	
    34	// TODO:
    35	// Add a -o flag that writes to the specified destination file.
    36	// Teach the script to create crt and key if -crt and -key aren't specified.
    37	var (
    38		crt = flag.String("crt", "", "path to nginx certificates.")
    39		key = flag.String("key", "", "path to nginx private key.")
    40	)
    41	
    42	func read(file string) []byte {
    43		b, err := ioutil.ReadFile(file)
    44		if err != nil {
    45			log.Fatalf("Cannot read file %v, %v", file, err)
    46		}
    47		return b
    48	}
    49	
    50	func main() {
    51		flag.Parse()
    52		if *crt == "" || *key == "" {
    53			log.Fatalf("Need to specify -crt -key and -template")
    54		}
    55		nginxCrt := read(*crt)
    56		nginxKey := read(*key)
    57		secret := &api.Secret{
    58			ObjectMeta: api.ObjectMeta{
    59				Name: "nginxsecret",
    60			},
    61			Data: map[string][]byte{
    62				"nginx.crt": nginxCrt,
    63				"nginx.key": nginxKey,
    64			},
    65		}
    66		fmt.Printf(runtime.EncodeOrDie(api.Codecs.LegacyCodec(registered.EnabledVersions()...), secret))
    67	}
