    96	func envInt(key string, def int) int {
    97		if env := os.Getenv(key); env != "" {
    98			val, err := strconv.Atoi(env)
    99			if err != nil {
   100				log.Printf("invalid value for %q: using default: %q", key, def)
   101				return def
   102			}
   103			return val
   104		}
   105		return def

   110	func main() {
   111		flag.Parse()
   112		if *flRepo == "" || *flDest == "" {
   113			flag.Usage()
   114			log.Fatal(usage)
   115		}
   116		if _, err := exec.LookPath("git"); err != nil {
   117			log.Fatalf("required git executable not found: %v", err)
   118		}
   119	
   120		if *flUsername != "" && *flPassword != "" {
   121			if err := setupGitAuth(*flUsername, *flPassword, *flRepo); err != nil {
   122				log.Fatalf("error creating .netrc file: %v", err)
   123			}
   124		}
   125	
   126		if *flSSH {
   127			if err := setupGitSSH(); err != nil {
   128				log.Fatalf("error configuring SSH: %v", err)
   129			}
   130		}
   131	
   132		initialSync := true
   133		failCount := 0
   134		for {
   135			if err := syncRepo(*flRepo, *flDest, *flBranch, *flRev, *flDepth); err != nil {
   136				if initialSync || failCount >= *flMaxSyncFailures {
   137					log.Fatalf("error syncing repo: %v", err)
   138				}
   139	
   140				failCount++
   141				log.Printf("unexpected error syncing repo: %v", err)
   142				log.Printf("waiting %d seconds before retryng", *flWait)
   143				time.Sleep(time.Duration(*flWait) * time.Second)
   144				continue
   145			}
   146	
   147			initialSync = false
   148			failCount = 0
   149	
   150			if *flOneTime {
   151				os.Exit(0)
   152			}
   153	
   154			log.Printf("waiting %d seconds", *flWait)
   155			time.Sleep(time.Duration(*flWait) * time.Second)
   156			log.Println("done")
   157		}
   158	}

   160	// updateSymlink atomically swaps the symlink to point at the specified directory and cleans up the previous worktree.
   161	func updateSymlink(link, newDir string) error {
   162		// Get currently-linked repo directory (to be removed), unless it doesn't exist
   163		currentDir, err := filepath.EvalSymlinks(path.Join(volMount, link))
   164		if err != nil && !os.IsNotExist(err) {
   165			return fmt.Errorf("error accessing symlink: %v", err)
   166		}
   167	
   168		// newDir is /git/rev-..., we need to change it to relative path.
   169		// Volume in other container may not be mounted at /git, so the symlink can't point to /git.
   170		newDirRelative, err := filepath.Rel(volMount, newDir)
   171		if err != nil {
   172			return fmt.Errorf("error converting to relative path: %v", err)
   173		}
   174	
   175		if _, err := runCommand("ln", volMount, []string{"-snf", newDirRelative, "tmp-link"}); err != nil {
   176			return fmt.Errorf("error creating symlink: %v", err)
   177		}
   178	
   179		log.Printf("create symlink %v->%v", "tmp-link", newDirRelative)
   180	
   181		if _, err := runCommand("mv", volMount, []string{"-T", "tmp-link", link}); err != nil {
   182			return fmt.Errorf("error replacing symlink: %v", err)
   183		}
   184	
   185		log.Printf("rename symlink %v to %v", "tmp-link", link)
   186	
   187		// Clean up previous worktree
   188		if len(currentDir) > 0 {
   189			if err = os.RemoveAll(currentDir); err != nil {
   190				return fmt.Errorf("error removing directory: %v", err)
   191			}
   192	
   193			log.Printf("remove %v", currentDir)
   194	
   195			output, err := runCommand("git", volMount, []string{"worktree", "prune"})
   196			if err != nil {
   197				return err
   198			}
   199	
   200			log.Printf("worktree prune %v", string(output))
   201		}
   202	
   203		return nil
   204	}

   206	// addWorktreeAndSwap creates a new worktree and calls updateSymlink to swap the symlink to point to the new worktree
   207	func addWorktreeAndSwap(dest, branch, rev string) error {
   208		// fetch branch
   209		output, err := runCommand("git", volMount, []string{"fetch", "origin", branch})
   210		if err != nil {
   211			return err
   212		}
   213	
   214		log.Printf("fetch %q: %s", branch, string(output))
   215	
   216		// add worktree in subdir
   217		rand.Seed(time.Now().UnixNano())
   218		worktreePath := path.Join(volMount, "rev-"+strconv.Itoa(rand.Int()))
   219		output, err = runCommand("git", volMount, []string{"worktree", "add", worktreePath, "origin/" + branch})
   220		if err != nil {
   221			return err
   222		}
   223	
   224		log.Printf("add worktree origin/%q: %v", branch, string(output))
   225	
   226		// .git file in worktree directory holds a reference to /git/.git/worktrees/<worktree-dir-name>
   227		// Replace it with a reference using relative paths, so that other containers can use a different volume mount name
   228		worktreePathRelative, err := filepath.Rel(volMount, worktreePath)
   229		if err != nil {
   230			return err
   231		}
   232		gitDirRef := []byte(path.Join("gitdir: ../.git/worktrees", worktreePathRelative) + "\n")
   233		if err = ioutil.WriteFile(path.Join(worktreePath, ".git"), gitDirRef, 0644); err != nil {
   234			return err
   235		}
   236	
   237		// reset working copy
   238		output, err = runCommand("git", worktreePath, []string{"reset", "--hard", rev})
   239		if err != nil {
   240			return err
   241		}
   242	
   243		log.Printf("reset %q: %v", rev, string(output))
   244	
   245		if *flChmod != 0 {
   246			// set file permissions
   247			_, err = runCommand("chmod", "", []string{"-R", string(*flChmod), worktreePath})
   248			if err != nil {
   249				return err
   250			}
   251		}
   252	
   253		return updateSymlink(dest, worktreePath)
   254	}

   256	func initRepo(repo, dest, branch, rev string, depth int) error {
   257		// clone repo
   258		args := []string{"clone", "--no-checkout", "-b", branch}
   259		if depth != 0 {
   260			args = append(args, "-depth")
   261			args = append(args, string(depth))
   262		}
   263		args = append(args, repo)
   264		args = append(args, volMount)
   265		log.Println("initRepo", args)
   266	
   267		output, err := runCommand("git", "", args)
   268		if err != nil {
   269			return err
   270		}
   271	
   272		log.Printf("clone %q: %s", repo, string(output))
   273	
   274		return nil
   275	}

   277	// syncRepo syncs the branch of a given repository to the destination at the given rev.
   278	func syncRepo(repo, dest, branch, rev string, depth int) error {
   279		target := path.Join(volMount, dest)
   280		gitRepoPath := path.Join(target, ".git")
   281		_, err := os.Stat(gitRepoPath)
   282		switch {
   283		case os.IsNotExist(err):
   284			err = initRepo(repo, target, branch, rev, depth)
   285			if err != nil {
   286				return err
   287			}
   288		case err != nil:
   289			return fmt.Errorf("error checking if repo exist %q: %v", gitRepoPath, err)
   290		default:
   291			needUpdate, err := gitRemoteChanged(target, branch)
   292			if err != nil {
   293				return err
   294			}
   295			if !needUpdate {
   296				log.Printf("No change")
   297				return nil
   298			}
   299		}
   300	
   301		return addWorktreeAndSwap(dest, branch, rev)
   302	}

   304	// gitRemoteChanged returns true if the remote HEAD is different from the local HEAD, false otherwise
   305	func gitRemoteChanged(localDir, branch string) (bool, error) {
   306		_, err := runCommand("git", localDir, []string{"remote", "update"})
   307		if err != nil {
   308			return false, err
   309		}
   310		localHead, err := runCommand("git", localDir, []string{"rev-parse", "HEAD"})
   311		if err != nil {
   312			return false, err
   313		}
   314		remoteHead, err := runCommand("git", localDir, []string{"rev-parse", fmt.Sprintf("origin/%v", branch)})
   315		if err != nil {
   316			return false, err
   317		}
   318		return (strings.Compare(string(localHead), string(remoteHead)) != 0), nil
   319	}

   321	func runCommand(command, cwd string, args []string) ([]byte, error) {
   322		cmd := exec.Command(command, args...)
   323		if cwd != "" {
   324			cmd.Dir = cwd
   325		}
   326		output, err := cmd.CombinedOutput()
   327		if err != nil {
   328			return []byte{}, fmt.Errorf("error running command %q : %v: %s", strings.Join(cmd.Args, " "), err, string(output))
   329		}
   330	
   331		return output, nil
   332	}

   334	func setupGitAuth(username, password, gitURL string) error {
   335		log.Println("setting up the git credential cache")
   336		cmd := exec.Command("git", "config", "--global", "credential.helper", "cache")
   337		output, err := cmd.CombinedOutput()
   338		if err != nil {
   339			return fmt.Errorf("error setting up git credentials %v: %s", err, string(output))
   340		}
   341	
   342		cmd = exec.Command("git", "credential", "approve")
   343		stdin, err := cmd.StdinPipe()
   344		if err != nil {
   345			return err
   346		}
   347		creds := fmt.Sprintf("url=%v\nusername=%v\npassword=%v\n", gitURL, username, password)
   348		io.Copy(stdin, bytes.NewBufferString(creds))
   349		stdin.Close()
   350		output, err = cmd.CombinedOutput()
   351		if err != nil {
   352			return fmt.Errorf("error setting up git credentials %v: %s", err, string(output))
   353		}
   354	
   355		return nil
   356	}

   358	func setupGitSSH() error {
   359		log.Println("setting up git SSH credentials")
   360	
   361		if _, err := os.Stat("/etc/git-secret/ssh"); err != nil {
   362			return fmt.Errorf("error: could not find SSH key Secret: %v", err)
   363		}
   364	
   365		// Kubernetes mounts Secret as 0444 by default, which is not restrictive enough to use as an SSH key.
   366		// TODO: Remove this command once Kubernetes allows for specifying permissions for a Secret Volume.
   367		// See https://github.com/kubernetes/kubernetes/pull/28936.
   368		if err := os.Chmod("/etc/git-secret/ssh", 0400); err != nil {
   369	
   370			// If the Secret Volume is mounted as readOnly, the read-only filesystem nature prevents the necessary chmod.
   371			return fmt.Errorf("error running chmod on Secret (make sure Secret Volume is NOT mounted with readOnly=true): %v", err)
   372		}
   373	
   374		return nil
   375	}
