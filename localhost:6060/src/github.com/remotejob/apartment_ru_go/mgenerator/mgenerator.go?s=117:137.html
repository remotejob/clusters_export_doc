    12	type Prefix []string
    13	
    14	// String returns the Prefix as a string (for use as a map key).
    15	func (p Prefix) String() string {
    16		return strings.Join(p, " ")
    17	}
    18	
    19	// Shift removes the first word from the Prefix and appends the given word.
    20	func (p Prefix) Shift(word string) {
    21		copy(p, p[1:])
    22		p[len(p)-1] = word
    23	}

    25	type Chain struct {
    26		chain     map[string][]string
    27		prefixLen int
    28	}
    29	
    30	func NewChain(prefixLen int) *Chain {
    31		return &Chain{make(map[string][]string), prefixLen}
    32	}
    33	
    34	func (c *Chain) Write(b []byte) {
    35		br := bytes.NewReader(b)
    36		p := make(Prefix, c.prefixLen)
    37		for {
    38			var s string
    39			if _, err := fmt.Fscan(br, &s); err != nil {
    40				break
    41			}
    42			key := p.String()
    43			c.chain[key] = append(c.chain[key], s)
    44			p.Shift(s)
    45		}
    46	}
    47	
    48	func (c *Chain) Generate(n int) string {
    49		p := make(Prefix, c.prefixLen)
    50		var words []string
    51		for i := 0; i < n; i++ {
    52			choices := c.chain[p.String()]
    53			if len(choices) == 0 {
    54				break
    55			}
    56			next := choices[rand.Intn(len(choices))]
    57	
    58			words = append(words, next)
    59	
    60			p.Shift(next)
    61		}
    62		return strings.Join(words, " ")
    63	}
