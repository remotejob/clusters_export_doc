    50	func (jo *JobOffer) ParsePage(urlstr string) {
    51	
    52		resp, err := http.Get(urlstr)
    53		if err != nil {
    54			panic(err)
    55		}
    56		root, err := html.Parse(resp.Body)
    57		if err != nil {
    58			panic(err)
    59	
    60		}
    61	
    62		matcher := func(n *html.Node) bool {
    63			// must check for nil values
    64			if n.DataAtom == atom.A {
    65	
    66				return true
    67	
    68			}
    69			return false
    70		}
    71	
    72		jo.FindLocation(root)
    73		jo.FindDescription(root)
    74	
    75		grid, ok := scrape.Find(root, scrape.ByClass("jobdetail"))
    76	
    77		if ok {
    78			gridItems := scrape.FindAll(grid, matcher)
    79			now := time.Now()
    80	
    81			for _, link := range gridItems {
    82	
    83				id := scrape.Attr(link, "id")
    84				data_uri := scrape.Attr(link, "data-uri")
    85				href := scrape.Attr(link, "href")
    86				data_email := scrape.Attr(link, "data-email")
    87	
    88				class := scrape.Attr(link, "class")
    89	
    90				text := scrape.Text(link)
    91	
    92				if class == "title job-link" {
    93	
    94					//				jo.Id = href
    95	
    96					u, err := url.Parse(urlstr)
    97					if err != nil {
    98						panic(err)
    99					}
   100	
   101					id_from_link := u.Scheme +"://"+ u.Host + u.Path
   102	
   103					jo.Id = id_from_link
   104					jo.Created_at = now
   105	
   106					if text != "" {
   107						jo.Title = text
   108					}
   109	
   110				}
   111	
   112				if class == "post-tag job-link no-tag-menu" {
   113					if text != "" {
   114						jo.Tags = append(jo.Tags, text)
   115					}
   116	
   117				}
   118				if class == "employer" {
   119					if text != "" {
   120						jo.Company = text
   121					}
   122	
   123				}
   124	
   125				if id == "apply" && data_uri != "" {
   126					jo.Externallink = data_uri
   127				}
   128	
   129				if id == "apply" && data_email != "" {
   130	
   131					jo.Externallink = data_email
   132	
   133				}
   134	
   135				if href != "" && text != "" {
   136	
   137					if strings.HasPrefix(href, "mailto") {
   138						jo.Email = text
   139					}
   140	
   141				}
   142	
   143			}
   144	
   145		}
   146	
   147	}
